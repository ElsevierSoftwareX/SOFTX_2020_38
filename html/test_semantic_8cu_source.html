197 * \fn void intreSemanticCPU(float *initiPop, float *OutSemantic, float *data, int nrow, int depth, int numIndi)
198 * \brief This function interprets the generated individuals in CPU to obtain their semantics.
199 * \param float *initiPop: This vector pointers to store the individuals of the initial population.
200 * \param flot *OutSemantic: vector pointers to store the semantics of each individual in the population.
201 * \param float *data: This pointer vector containing training or test data.
202 * \param int *nrow: This variable contains the number of fitness cases.
203 * \param int depth: This variable thar stores maximum depth for individuals
204 * \param int numIndi: This variable contains the number of individuals that exist in the initial population.
205 * \return void
206 * \date 05/12/2020
207 * \author José Manuel Muñoz Contreras, Leonardo Trujillo, Daniel E. Hernandez, Perla Juárez Smith
208 * \file testSemantic.cu
209 */
210 void intreSemanticCPU(float *initiPop, float *OutSemantic, float *data, int nrow, int depth, int numIndi){
211  float tmp,tmp2,out;
212  stack <float> a;
213  for (int l=0; l<nrow; l++) {
214  out=0;
215  clean(a);
216  a = stack<float>();
217  clear(a);
218  if (a.empty()) {
219  for (int i=0; i<numIndi; i++) {
220  for (int j=0; j<depth; j++) {
221  if (initiPop[i*depth+j]>0) {
222  a.push(initiPop[i*depth+j]);
223  }else if (initiPop[i*depth+j] <= -1000) {
224  a.push(data[l]);
225  } else if (initiPop[i*depth+j]== -1) {
226  if (!a.empty()) {
227  tmp = a.top();
228  a.pop();
229  } if (!a.empty()) {
230  tmp2 = a.top();
231  a.pop();
232  a.push(tmp+tmp2);
233  out=tmp+tmp2;
234  } else {
235  a.push(tmp);
236  }
237  } else if (initiPop[i*depth+j]== -2) {
238  if (!a.empty()) {
239  tmp = a.top();
240  a.pop();
241  } if (!a.empty()) {
242  tmp2 = a.top();
243  a.pop();
244  out=tmp-tmp2;
245  a.push(out);
246  } else {
247  a.push(tmp);
248  }
249  } else if (initiPop[i*depth+j]== -3) {
250  if (!a.empty()) {
251  tmp = a.top();
252  a.pop();
253  } if (!a.empty()) {
254  tmp2 = a.top();
255  a.pop();
256  out=tmp*tmp2;
257  a.push(out);

259  } else {
260  a.push(tmp);
261  }
262  } else if (initiPop[i*depth+j]== -4) {
263  if (!a.empty()) {
264  tmp = a.top();
265  a.pop();
266  } if (!a.empty()) {
267  tmp2 = a.top();
268  a.pop();
