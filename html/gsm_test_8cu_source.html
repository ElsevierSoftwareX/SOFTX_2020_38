81 /*!
82 * \fn void omsCPU(float *initialPopSemantic, float *randomTreesSemantic, float *newOffspringSemantic, int sizePopulation ,int nrow, double ms, float *index)
83 * \brief The GSM operator is basically a vector addition operation, that can be performed independently for each semantic element STi,j.
84  However, it is necessary to select the semantics of two random trees R u and R v , and a random mutation step ms.
85 * \param float *initialPopSemantic: This vector of pointers contains the semantics of the initial population
86 * \param float *randomTreesSemantic: This vector of pointers contains the semantics of the random trees
87 * \param float *newOffspringSemantic: This vector of pointers will store the semantics of the new offspring
88 * \param int sizePopulation: This variable contains the number of individuals that the population has
89 * \param int nrow: Variable containing the number of rows (instances) of the training dataset
90 * \param double ms: This variable stores a random value that causes a disturbance in the sematics of an individual
91 * \param int generation: Number of generation
92 * \param float *indexRandomTrees: This pointer stores the indexes randomly for mutation
93 * \return void
94 * \date 05/12/2020
95 * \author José Manuel Muñoz Contreras, Leonardo Trujillo, Daniel E. Hernandez, Perla Juárez Smith
96 * \file omsTest.cu
97 */
98 void omsCPU(float *initialPopSemantic, float *randomTreesSemantic, float *newOffspringSemantic, int sizePopulation ,int nrow, double ms, float *index){
99  int seed;
100  int firstTree, secondTree;
101  for (int i=0; i<sizePopulation; i++) {
102  for (int j=0; j<nrow; j++) {
103  seed = i+j/nrow;
104  firstTree = index[seed];
105  secondTree = index[sizePopulation + seed];
106  float sigmoid_1=1.0/(1+std::exp(-(randomTreesSemantic[firstTree*nrow+j])));
107  float sigmoid_2=1.0/(1+std::exp(-(randomTreesSemantic[secondTree*nrow+j])));
108  newOffspringSemantic[i*nrow+j] = initialPopSemantic[i*nrow+j] + (ms *(sigmoid_1-sigmoid_2));
109  }
110  }
111 }
