226 __global__ void computeSemantics(float *inputPopulation, float *outSemantic, unsigned int sizeMaxDepthIndividual, float *data,
227  int nrow, int nvar, int *pushGenes, float *stackInd)
228 {
229 
230  const unsigned int tid = threadIdx.x+blockIdx.x*blockDim.x;
231  pushGenes[tid] = tid * sizeMaxDepthIndividual;
232  int t,t_;
233  float tmp,tmp2,out;
234  for(int k=0; k<nrow; k++)
235  {
236  out=0;
237  clearStack(pushGenes,sizeMaxDepthIndividual, stackInd);
238  for(int i=0; i < sizeMaxDepthIndividual; i++)
239  {
240 
241  if(inputPopulation[tid*sizeMaxDepthIndividual+i] > 0)
242  {
243  pushGenes[tid] = push(inputPopulation[tid*sizeMaxDepthIndividual+i],pushGenes,stackInd);
244  }
245  else
246  if(inputPopulation[tid*sizeMaxDepthIndividual+i] <= -1000)
247  {
248  t=inputPopulation[tid*sizeMaxDepthIndividual+i];
249  t_=(t+1000)*(-1);
250  pushGenes[tid] = push(data[t_+nvar*k],pushGenes,stackInd);
251  }
252  else
253  if (inputPopulation[tid*sizeMaxDepthIndividual+i] == -1)
254  {
255  if (!isEmpty(pushGenes,sizeMaxDepthIndividual))
256  {
257  tmp = pop(pushGenes,stackInd);
258  if (!isEmpty(pushGenes,sizeMaxDepthIndividual))
259  {
260  tmp2 = pop(pushGenes,stackInd);
261  if (!isnan(tmp) || !isnan(tmp2))
262  {
263  pushGenes[tid] = push(tmp2 + tmp,pushGenes,stackInd);
264  out = tmp2+tmp;
265  }
266  }
267  else
268  pushGenes[tid] = push(tmp,pushGenes,stackInd);
269  }
270  }
271  else
272  if (inputPopulation[tid*sizeMaxDepthIndividual+i] == -2)
273  {
274  if(!isEmpty(pushGenes,sizeMaxDepthIndividual))
275  {
276  tmp = pop(pushGenes,stackInd);
277  if (!isEmpty(pushGenes,sizeMaxDepthIndividual))
278  {
279  tmp2 = pop(pushGenes,stackInd);
280  if (!isnan(tmp) || !isnan(tmp2))
281  {
282  pushGenes[tid] = push(tmp2 - tmp,pushGenes,stackInd);
283  out = tmp2-tmp;
284  }
285  }
286  }else
287  pushGenes[tid] = push(tmp,pushGenes,stackInd);
288  }
289  else
290  if (inputPopulation[tid*sizeMaxDepthIndividual+i] == -3)
291  {
292  if (!isEmpty(pushGenes,sizeMaxDepthIndividual))
293  {
294  tmp = pop(pushGenes,stackInd);
295  if (!isEmpty(pushGenes,sizeMaxDepthIndividual))
296  {
297  tmp2 = pop(pushGenes,stackInd);
298  if (!isnan(tmp) || !isnan(tmp2))
299  {
300  pushGenes[tid] = push(tmp2 * tmp,pushGenes,stackInd);
301  out = tmp2*tmp;
302  }
303  }
304  }else
305  pushGenes[tid] = push(tmp,pushGenes,stackInd);
306  }
307  else
308  if (inputPopulation[tid*sizeMaxDepthIndividual+i] == -4)
309  {
310  if (!isEmpty(pushGenes,sizeMaxDepthIndividual))
311  {
312  tmp = pop(pushGenes,stackInd);
313  if (!isEmpty(pushGenes,sizeMaxDepthIndividual))
314  {
315  tmp2 = pop(pushGenes,stackInd);
316  if (!isnan(tmp) || !isnan(tmp2))
317  {
318  pushGenes[tid] = push(tmp2 / sqrtf(1+tmp*tmp),pushGenes,stackInd);
319  out = tmp2 / sqrtf(1+tmp*tmp);
320  }
321  }
322  }else
323  pushGenes[tid] = push(tmp,pushGenes,stackInd);
324  }
325  }
326  outSemantic[(tid*nrow+k)] = out;
327  }
328 }
