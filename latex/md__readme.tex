\hypertarget{md__readme_autotoc_md4}{}\doxysection{Compilation requirements, operating enviroments \& dependencies}\label{md__readme_autotoc_md4}

\begin{DoxyCode}{0}
\DoxyCodeLine{Toolkit CUDA v10.1 \&\& v9.2, GCC v7.4.0, CUBLAS v2.0, Linux Headers, unix-\/like systems, Ubuntu Linux18.04}
\DoxyCodeLine{}
\DoxyCodeLine{How to compile.}
\DoxyCodeLine{}
\DoxyCodeLine{nvcc -\/std=c++11 -\/O0 GsgpCuda.cu -\/o GsgpCuda.x  -\/lcublas}
\DoxyCodeLine{}
\DoxyCodeLine{To run gsgpCuda it is necessary to add a name for the output file generation, as shown in the example.}
\DoxyCodeLine{}
\DoxyCodeLine{./GsgpCuda.x -\/train\_data train.txt -\/test\_data test.txt -\/output\_model best\_model}
\DoxyCodeLine{}
\DoxyCodeLine{train.txt: This file must contain the training data.}
\DoxyCodeLine{test.txt:  This file must contain the test data.}
\DoxyCodeLine{best\_model: This file contains the information needed to test the model generated by GsgpCuda.}
\DoxyCodeLine{    best\_model\_initialPopulation.csv: This file will store the individuals of the initial population.}
\DoxyCodeLine{    best\_model\_randomTrees.csv: This file will store the individuals of the auxiliary population.}
\DoxyCodeLine{    best\_model\_fitnessTrain.csv: This file will store the error of the best individual in each generation with training data.}
\DoxyCodeLine{    best\_model\_fitnessTest.csv: This file will store the error of the best individual in each generation with test data.}
\DoxyCodeLine{    best\_model\_processing\_time.csv: This file stores the processing times in seconds of the various modules of the algorithm. }
\end{DoxyCode}

\hypertarget{md__readme_autotoc_md5}{}\doxysection{How to test best model}\label{md__readme_autotoc_md5}

\begin{DoxyCode}{0}
\DoxyCodeLine{To test the model generated by GsgpCuda it is necessary to provide the name of the model by command line, }
\DoxyCodeLine{the second parameter indicates the name of the data file, the third parameter indicates the name of the file to save the output values generated by the model.}
\DoxyCodeLine{}
\DoxyCodeLine{./GsgpCuda.x -\/model best\_model -\/input\_data test.txt -\/prediction\_output predic\_values}
\DoxyCodeLine{}
\DoxyCodeLine{best\_model: This file contains the information needed to test the model generated by GsgpCuda.}
\DoxyCodeLine{test.txt: This file must contain the unknown data for testing the model.}
\DoxyCodeLine{predic\_values: This file will store the output values generated by the model when tested on unknown data.}
\end{DoxyCode}
