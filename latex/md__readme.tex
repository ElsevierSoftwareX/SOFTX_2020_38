\hypertarget{md__readme_autotoc_md4}{}\doxysection{Compilation requirements, operating enviroments \& dependencies}\label{md__readme_autotoc_md4}

\begin{DoxyCode}{0}
\DoxyCodeLine{Toolkit CUDA v10.1 \&\& v9.2, GCC v7.4.0, CUBLAS v2.0, Linux Headers, unix-\/like systems, Ubuntu Linux18.04}
\DoxyCodeLine{}
\DoxyCodeLine{How to compile.}
\DoxyCodeLine{}
\DoxyCodeLine{nvcc -\/std=c++11 -\/O0 GsgpCuda.cu -\/o GsgpCuda.x  -\/lcublas}
\DoxyCodeLine{}
\DoxyCodeLine{To run gsgpCuda it is necessary to add a name for the output file generation, as shown in the example.}
\DoxyCodeLine{}
\DoxyCodeLine{./GsgpCuda.x -\/train\_data <train\_file\_name>.txt -\/test\_data <test\_file\_name>.txt -\/output\_model <model\_name>}
\DoxyCodeLine{}
\DoxyCodeLine{<train\_file\_name>.txt: This file must contain the training data used to compute fitness and given in a format of space-\/seperated values with n columns, where the first n-\/1 columns are the input features and the last column is the target variable.}
\DoxyCodeLine{<test\_file\_name>.txt:  This file must contain the test data used to evalaute the best individual at each generation. this data does not influence the evolutionary/training process, and must be given in the same format as the training data.}
\DoxyCodeLine{If no test data is available, the user can indicate the same file as the training data to assure proper excecution, this does not afffect model evolution.}
\DoxyCodeLine{<model\_name>.csv: This file contains the information needed to apply the best model found by gsgpcuda on new data (make new predictions or inferences). other auxiliary files are also generated and required.}
\DoxyCodeLine{}
\DoxyCodeLine{GsgpCuda generates the following output files, which are located in the log folder}
\DoxyCodeLine{}
\DoxyCodeLine{    <model\_name>\_initialPopulation.csv: This file will store the individuals of the initial population.}
\DoxyCodeLine{    <model\_name>\_randomTrees.csv: This file will store the individuals of the auxiliary population.}
\DoxyCodeLine{    <model\_name>\_fitnessTrain.csv: This file will store the error of the best individual in each generation with training data.}
\DoxyCodeLine{    <model\_name>\_fitnessTest.csv: This file will store the error of the best individual in each generation with test data.}
\DoxyCodeLine{    <model\_name>\_processing\_time.csv: This file stores the processing times in seconds of the various modules of the algorithm. }
\end{DoxyCode}

\hypertarget{md__readme_autotoc_md5}{}\doxysection{How to make new predictions (inference) with the best model found}\label{md__readme_autotoc_md5}

\begin{DoxyCode}{0}
\DoxyCodeLine{To make new predictions with the best model generated by gsgpcuda it is necessary to provide the name of the model by command line, the second parameter indicates the name of the data file, and the third parameter indicates the name of the file to save the output values generated by the model.}
\DoxyCodeLine{}
\DoxyCodeLine{./GsgpCuda.x -\/model <model\_name> -\/input\_data <new\_data>.txt -\/prediction\_output <predicted\_values>.csv}
\DoxyCodeLine{}
\DoxyCodeLine{<model\_name>: this file contains the information needed to test the model generated by gsgpcuda.}
\DoxyCodeLine{<new\_data>.txt: this file must contain the new unseen data for testing the model. the file should have the same number of columns as the number of input features in the train.txt file used to train the data, without the target column.}
\DoxyCodeLine{<predicted\_values>.csv: this file will store the output values generated by the model when tested on the data in newdata.txt.}
\end{DoxyCode}
